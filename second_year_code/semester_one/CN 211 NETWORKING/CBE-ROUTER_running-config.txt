enable
conf t
hostname CBE-ROUTER

interface GigabitEthernet0/2
 description WAN to Cloud
 ip address 200.0.0.5 255.255.255.0
 ip nat outside
 no shutdown
exit

interface GigabitEthernet0/0
 description Server Network
 ip address 192.168.11.1 255.255.255.0
 ip nat inside
 no shutdown
exit

interface GigabitEthernet0/1
 description Users LAN to Switch
 ip address 192.168.10.1 255.255.255.0
 ip nat inside
 no shutdown
exit

! DHCP for USERS only
ip dhcp excluded-address 192.168.10.1 192.168.10.20

ip dhcp pool CBE_USERS
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 192.168.11.2
exit

! NAT overload (both inside nets)
access-list 1 permit 192.168.10.0 0.0.0.255
access-list 1 permit 192.168.11.0 0.0.0.255
ip nat inside source list 1 interface GigabitEthernet0/2 overload

! Publish web server
ip nat inside source static tcp 192.168.11.2 80 200.0.0.5 80

! Default route
ip route 0.0.0.0 0.0.0.0 200.0.0.1

end
wr
